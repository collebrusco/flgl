cmake_minimum_required(VERSION 3.10)
project(ExampleProject)

# Specify the C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Add your OpenGL library as a subdirectory
add_subdirectory(lib/flgl)

# Include directories for your project
include_directories(${PROJECT_SOURCE_DIR}/lib/flgl/inc)

# Define the source files for your project
file(GLOB_RECURSE PROJECT_SOURCES "src/*.cpp")

# Define the executable
add_executable(${PROJECT_NAME} ${PROJECT_SOURCES})

# Link your project with your OpenGL library
target_link_libraries(${PROJECT_NAME} FLGL)
